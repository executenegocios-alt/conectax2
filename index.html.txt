    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // 1. Inicializa Ícones Lucide
            lucide.createIcons();

            // 2. Define o ano atual no rodapé
            document.getElementById('current-year').textContent = new Date().getFullYear();

            // 3. Lógica do Menu Mobile
            const menuButton = document.getElementById('menu-button');
            const mobileMenu = document.getElementById('mobile-menu');
            
            menuButton.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            // Fecha menu mobile ao clicar em um link
            mobileMenu.querySelectorAll('a').forEach(link => {
                link.addEventListener('click', () => {
                    mobileMenu.classList.add('hidden');
                });
            });

            // 4. Simulação de Envio de Formulário de Contato
            const contactForm = document.getElementById('contact-form');
            const formMessage = document.getElementById('form-message');

            contactForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                // Limpa o estado anterior
                formMessage.classList.add('hidden');
                formMessage.classList.remove('bg-red-100', 'text-red-800', 'bg-green-100', 'text-green-800');
                
                // Simulação de delay de envio
                setTimeout(() => {
                    // Simula sucesso
                    formMessage.textContent = 'Mensagem enviada com sucesso! Em breve entraremos em contato.';
                    formMessage.classList.remove('hidden');
                    formMessage.classList.add('bg-green-100', 'text-green-800');
                    contactForm.reset();
                    
                    // Oculta a mensagem após 5 segundos
                    setTimeout(() => {
                        formMessage.classList.add('hidden');
                    }, 5000);

                }, 800);
            });

            // 5. Scroll Reveal (Animações Suaves)
            // Função para verificar se o elemento está visível no viewport
            const isElementInViewport = (el) => {
                const rect = el.getBoundingClientRect();
                // Verifica se o elemento está pelo menos parcialmente visível
                return (
                    rect.top <= (window.innerHeight || document.documentElement.clientHeight) * 0.9 &&
                    rect.bottom >= 0
                );
            };

            // Adiciona classe de animação aos itens
            const revealItems = document.querySelectorAll('.reveal-item');
            const handleScroll = () => {
                revealItems.forEach(item => {
                    if (isElementInViewport(item)) {
                        item.classList.add('is-visible');
                    }
                });
            };

            // Adiciona listener de scroll e executa na carga
            window.addEventListener('scroll', handleScroll);
            handleScroll(); 
        });
    </script>
